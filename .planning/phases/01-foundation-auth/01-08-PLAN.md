```python
    await limiter.acquire(1)  # Should block until refill
```

```python
"""
Import guard — ensures no module in app/agents/ or app/graph/ imports groq directly.
This is the CI enforcement for the Groq rate limiter boundary (per D-20, SPEC Requirement 6).
"""
import sys
import importlib
import pkgutil
import pytest


def _import_all_submodules(package_name: str) -> None:
    """Import all submodules of a package by name."""
    try:
        package = importlib.import_module(package_name)
    except ModuleNotFoundError:
        return  # Package doesn't exist yet (Phase 1 agents/ and graph/ are empty)
    
    package_path = getattr(package, "__path__", None)
    if package_path is None:
        return
    
    for importer, module_name, is_pkg in pkgutil.walk_packages(
        path=package_path, prefix=package_name + ".", onerror=lambda name: None
    ):
        try:
            importlib.import_module(module_name)
        except Exception:
            pass  # Import failures don't hide boundary violations


def test_no_groq_import_in_agents():
    """app/agents/ modules must NOT import groq directly."""
    # Remove groq from sys.modules if already imported to get a clean slate
    sys.modules.pop("groq", None)
    
    _import_all_submodules("app.agents")
    
    assert "groq" not in sys.modules, (
        "BOUNDARY VIOLATION: A module in app/agents/ imported 'groq' directly. "
        "All Groq calls must go through app.services.groq_client.groq_rate_limiter."
    )


def test_no_groq_import_in_graph():
    """app/graph/ modules must NOT import groq directly."""
    sys.modules.pop("groq", None)
    
    _import_all_submodules("app.graph")
    
    assert "groq" not in sys.modules, (
        "BOUNDARY VIOLATION: A module in app/graph/ imported 'groq' directly. "
        "All Groq calls must go through app.services.groq_client.groq_rate_limiter."
    )
```
